## Example Discord Bot

```js title="index.js" frame="code"
require("dotenv").config();

const { Client } = require("oceanic.js");
const client = new Client({
  auth: `Bot ${process.env.DISCORD_TOKEN}`,
  gateway: {
    intents: ["GUILDS"],
  },
});

client.connect();

client.on("ready", () => {
  console.log(`${client.user.username} has been connected`);
});

client.on("error", (error) => {
  console.error(error);
});
```
